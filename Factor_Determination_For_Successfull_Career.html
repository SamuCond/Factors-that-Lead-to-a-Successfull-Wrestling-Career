# Create function that will extract Fighter Information from their Overview page

overview_info<- function(html_content) {
  html <- read_html(url(html_content))
  info <- html %>% 
    html_elements(".InformationBoxRow") %>% 
    html_text(trim = TRUE)
# Not all fighters have the same information, so we will filter the information and give NA where they dont have it
  keywords <- c("Birthplace", "Gender", "Height", "Weight", "Background in sports", "In-ring experience", "Wrestling style")
# Initialize an empty dataframe
  df <- data.frame(matrix(ncol = length(keywords), nrow = 1))
  names(df) <- keywords
# Use a for loop to extract the info and assign NA values if the info is not present
  for (i in keywords) {
    pattern <- paste0(i, ":")
    matched_info <- grep(pattern, info, value = TRUE)
    value <- ifelse(length(matched_info) > 0, sub(".*: ", "", matched_info), NA)
    df[1, i] <- value
  }
  return(df)
}


# Apply function
